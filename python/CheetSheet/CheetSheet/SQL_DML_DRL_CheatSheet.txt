SQL DML / DRL CHEAT SHEET
(Focus: SELECT Statements)

1Ô∏è‚É£ BASIC SELECT
------------------------------
SELECT column1, column2 FROM table_name;
SELECT * FROM Sales;

2Ô∏è‚É£ FILTERING DATA (WHERE)
------------------------------
SELECT * FROM Sales WHERE Quantity > 10;

Operators:
=, != or <>, >, <, >=, <=
BETWEEN ... AND ...
IN (...)
LIKE 'A%'
IS NULL / IS NOT NULL

Examples:
SELECT * FROM Sales WHERE Product = 'Apple';
SELECT * FROM Sales WHERE Quantity BETWEEN 5 AND 10;
SELECT * FROM Sales WHERE Product IN ('Apple','Banana');
SELECT * FROM Sales WHERE Product LIKE 'A%';

3Ô∏è‚É£ COMBINE CONDITIONS
------------------------------
SELECT * FROM Sales WHERE Quantity > 5 AND Price < 100;
SELECT * FROM Sales WHERE Product = 'Apple' OR Product = 'Banana';

4Ô∏è‚É£ SORTING RESULTS (ORDER BY)
------------------------------
SELECT Product, Total FROM Sales ORDER BY Total DESC;
ASC = ascending (default)
DESC = descending

5Ô∏è‚É£ LIMIT RESULTS
------------------------------
SELECT * FROM Sales ORDER BY Total DESC LIMIT 5;

6Ô∏è‚É£ AGGREGATE FUNCTIONS
------------------------------
COUNT() ‚Äì number of rows
SUM() ‚Äì total of values
AVG() ‚Äì average
MIN() ‚Äì minimum value
MAX() ‚Äì maximum value

Examples:
SELECT COUNT(*) FROM Sales;
SELECT SUM(Total) FROM Sales;
SELECT AVG(Price) FROM Sales;

7Ô∏è‚É£ GROUPING DATA (GROUP BY)
------------------------------
SELECT Product, SUM(Quantity) AS TotalQuantity
FROM Sales
GROUP BY Product;

8Ô∏è‚É£ FILTER GROUPS (HAVING)
------------------------------
SELECT Product, SUM(Total) AS TotalSales
FROM Sales
GROUP BY Product
HAVING SUM(Total) > 5000;

9Ô∏è‚É£ JOINS
------------------------------
INNER JOIN ‚Äì matching rows from both tables
LEFT JOIN ‚Äì all from left + matched from right

Examples:
SELECT * FROM Orders INNER JOIN Customers
ON Orders.CustomerID = Customers.CustomerID;

SELECT * FROM Orders LEFT JOIN Customers
ON Orders.CustomerID = Customers.CustomerID;

üîü ALIASES
------------------------------
SELECT Product AS Item, SUM(Total) AS Sales
FROM Sales AS S
GROUP BY S.Product;

1Ô∏è‚É£1Ô∏è‚É£ SUBQUERIES
------------------------------
SELECT Product, Total
FROM Sales
WHERE Total > (SELECT AVG(Total) FROM Sales);

1Ô∏è‚É£2Ô∏è‚É£ DATE FUNCTIONS (SQLite)
------------------------------
strftime('%Y', Date) ‚Äì Year
strftime('%m', Date) ‚Äì Month
strftime('%d', Date) ‚Äì Day
date('now') ‚Äì Current Date

Examples:
SELECT SUM(Total) FROM Sales WHERE strftime('%Y', Date)='2023';

1Ô∏è‚É£3Ô∏è‚É£ DISTINCT & ALIASES
------------------------------
SELECT DISTINCT Product FROM Sales;
SELECT Product AS ItemName, Price AS UnitPrice FROM Sales;

1Ô∏è‚É£4Ô∏è‚É£ COMMON QUERIES
------------------------------
Total Sales in 2023:
SELECT SUM(Total) FROM Sales WHERE strftime('%Y', Date)='2023';

Total Quantity per Product:
SELECT Product, SUM(Quantity) FROM Sales GROUP BY Product;

Highest Selling Product:
SELECT Product, SUM(Total) AS TotalSales
FROM Sales
GROUP BY Product
ORDER BY TotalSales DESC
LIMIT 1;

Monthly Sales Trend:
SELECT strftime('%m', Date) AS Month, SUM(Total) AS MonthlySales
FROM Sales
GROUP BY Month
ORDER BY Month;

Products with Average Price > 50:
SELECT Product, AVG(Price) AS AvgPrice
FROM Sales
GROUP BY Product
HAVING AvgPrice > 50;

1Ô∏è‚É£5Ô∏è‚É£ COMBINING MULTIPLE SELECTS
------------------------------
UNION ‚Äì combine unique results
UNION ALL ‚Äì combine with duplicates

Example:
SELECT Product FROM Sales2023 UNION SELECT Product FROM Sales2024;
SELECT Product FROM Sales2023 UNION ALL SELECT Product FROM Sales2024;
